<!DOCTYPE html>

<html lang="en">

<head>

  <meta charset="UTF-8">

  <title>Twitch Redirect</title>

</head>

<body>

<script>

  // Twitch redirects here with ?code=... or ?error=...

  const params = new URLSearchParams(window.location.search);

  const code = params.get("code");

  const error = params.get("error");


  if (code) {

    // Redirect back into your app via custom URL scheme

    window.location = `aircasttwitch://auth?code=${code}`;

  } else if (error) {

    window.location = `aircasttwitch://auth?error=${error}`;

  } else {

    document.body.innerHTML = "<p>Missing 'code' or 'error' query param.</p>";

  }

</script>

</body>

</html>
